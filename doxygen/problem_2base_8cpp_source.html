537 void base::objfun(fitness_vector &f, const decision_vector &x) const
538 {
539  // Some checks on the input values.
540  if (f.size() != m_f_dimension) {
541  pagmo_throw(value_error,"wrong fitness vector size when calling objective function");
542  }
543  if (x.size() != get_dimension()) {
544  pagmo_throw(value_error,"wrong decision vector size when calling objective function");
545  }
546  // Look into the cache.
547  typedef decision_vector_cache_type::iterator x_iterator;
548  typedef fitness_vector_cache_type::iterator f_iterator;
549  const x_iterator x_it = std::find(m_decision_vector_cache_f.begin(),m_decision_vector_cache_f.end(),x);
550  if (x_it == m_decision_vector_cache_f.end()) {
551  // Fitness is not into memory. Calculate it.
552  objfun_impl(f,x);
553  // Make sure that the implementation of objfun_impl() in the derived class did not fuck up the dimension of the fitness vector.
554  if (f.size() != m_f_dimension) {
555  pagmo_throw(value_error,"fitness dimension was changed inside objfun_impl()");
556  }
557  // Store the decision vector and the newly-calculated fitness in the front of the buffers.
558  m_decision_vector_cache_f.push_front(x);
559  m_fitness_vector_cache.push_front(f);
560  } else {
561  // Compute the corresponding iterator in the fitness vector cache.
562  f_iterator f_it = m_fitness_vector_cache.begin();
563  std::advance(f_it,std::distance(m_decision_vector_cache_f.begin(),x_it));
564  pagmo_assert(f_it != m_fitness_vector_cache.end());
565  // Assign to the fitness vector the value in the cache.
566  f = *f_it;
567  // Move the content of the current positions to the front of the buffers, and shift everything else down
568  // by one position.
569  x_iterator tmp_x_it = m_decision_vector_cache_f.begin();
570  f_iterator tmp_f_it = m_fitness_vector_cache.begin();
571  while (x_it != tmp_x_it) {
572  x_it->swap(*tmp_x_it);
573  f_it->swap(*tmp_f_it);
574  ++tmp_x_it;
575  ++tmp_f_it;
576  }
577  pagmo_assert(tmp_f_it == f_it);
578  }
579 }

610 bool base::compare_fitness_impl(const fitness_vector &v_f1, const fitness_vector &v_f2) const
611 {
612  pagmo_assert(v_f1.size() == v_f2.size()); // NOTE: v_f1.size CAN be different from m_f_dimension (i.e. when its a base_meta calling it)
613  f_size_type count1 = 0; f_size_type count2 = 0;
614  for (f_size_type i = 0; i < v_f1.size(); ++i) {
615  if (v_f1[i] < v_f2[i]) {
616  ++count1;
617  }
618  if (v_f1[i] == v_f2[i]) {
619  ++count2;
620  }
621  }
622  return ( ( (count1+count2) == v_f1.size()) && (count1>0) );
623 }

792 bool base::compare_fc_impl(const fitness_vector &f1, const constraint_vector &c1, const fitness_vector &f2, const constraint_vector &c2) const
793 {
794  const bool test1 = feasibility_c(c1), test2 = feasibility_c(c2);
795  if (test1 && !test2) {
796  return true;
797  }
798  if (!test1 && test2) {
799  return false;
800  }
801  // At this point, either they are both feasible or they are not.
802  if (test1) {
803  pagmo_assert(test2);
804  // If they are feasible, compare fitnesses and return.
805  return compare_fitness_impl(f1,f2);
806  } else {
807  pagmo_assert(!test2);
808  // If they are not feasible, compare constraints and return.
809  return compare_constraints_impl(c1,c2);
810  }
811 }

845 void base::compute_constraints(constraint_vector &c, const decision_vector &x) const
846 {
847  if (x.size() != get_dimension() || c.size() != get_c_dimension()) {
848  pagmo_throw(value_error,"invalid constraint and/or decision vector(s) size(s) during constraint testing");
849  }
850  // Do not do anything if constraints size is 0.
851  if (!m_c_dimension) {
852  return;
853  }
854  // Look into the cache.
855  typedef decision_vector_cache_type::iterator x_iterator;
856  typedef constraint_vector_cache_type::iterator c_iterator;
857  const x_iterator x_it = std::find(m_decision_vector_cache_c.begin(),m_decision_vector_cache_c.end(),x);
858  if (x_it == m_decision_vector_cache_c.end()) {
859  // Constraint vector is not into memory. Calculate it.
860  compute_constraints_impl(c,x);
861  // Make sure c was not fucked up in the implementation of constraints calculation.
862  if (c.size() != get_c_dimension()) {
863  pagmo_throw(value_error,"constraints dimension was changed inside compute_constraints_impl()");
864  }
865  // Store the decision vector and the newly-calculated constraint vector in the front of the buffers.
866  m_decision_vector_cache_c.push_front(x);
867  m_constraint_vector_cache.push_front(c);
868  } else {
869  // Compute the corresponding iterator in the constraint vector cache.
870  c_iterator c_it = m_constraint_vector_cache.begin();
871  std::advance(c_it,std::distance(m_decision_vector_cache_c.begin(),x_it));
872  pagmo_assert(c_it != m_constraint_vector_cache.end());
873  // Assign to the fitness vector the value in the cache.
874  c = *c_it;
875  // Move the content of the current positions to the front of the buffers, and shift everything else down
876  // by one position.
877  x_iterator tmp_x_it = m_decision_vector_cache_c.begin();
878  c_iterator tmp_c_it = m_constraint_vector_cache.begin();
879  while (x_it != tmp_x_it) {
880  x_it->swap(*tmp_x_it);
881  c_it->swap(*tmp_c_it);
882  ++tmp_x_it;
883  ++tmp_c_it;
884  }
885  pagmo_assert(tmp_c_it == c_it);
886  }
887 }

1004  // Equality constraints.

1007  ++count1;

1009  if (test_constraint(c2,i)) {
1010  ++count2;
1011  }
1012  norm1 += std::abs(c1[i]) * std::abs(c1[i]);

1014  }
1015  // Inequality constraints.
1016  for (c_size_type i = m_c_dimension - m_ic_dimension; i < m_c_dimension; ++i) {

1019  } else {

1021  }
1022  if (test_constraint(c2,i)) {
1023  ++count2;
1024  } else {
1025  norm2 += c2[i] * c2[i];
1026  }
1027  }
1028  if (count1 > count2) {
1029  return true;
1030  } else if (count1 < count2) {
1031  return false;
1032  } else {

1096 // This function will round to the nearest integer the upper/lower bounds of the integer part of the problem.
1097 // This should be called each time bounds are set.
1098 void base::normalise_bounds()
1099 {
1100  pagmo_assert(m_lb.size() >= m_i_dimension);
1101  // Flag to be set if we had to fix the bounds.
1102  bool bounds_fixed = false;
1103  for (size_type i = 0; i < m_lb.size() - m_i_dimension; ++i) {
1104  // Handle NaNs: if either lower/upper bound(s) are NaN, replace with 0 and 1 respectively.
1105  if (boost::math::isnan(m_lb[i]) || boost::math::isnan(m_ub[i])) {
1106  m_lb[i] = 0;
1107  m_ub[i] = 1;
1108  bounds_fixed = true;
1109  }
1110  // +-Infs are replaced by the highest/lowest values representable by double.
1111  if (boost::math::isinf(m_lb[i])) {
1112  m_lb[i] = (m_lb[i] > 0) ? boost::numeric::bounds<double>::highest() : boost::numeric::bounds<double>::lowest();
1113  bounds_fixed = true;
1114  }
1115  if (boost::math::isinf(m_ub[i])) {
1116  m_ub[i] = (m_ub[i] > 0) ? boost::numeric::bounds<double>::highest() : boost::numeric::bounds<double>::lowest();
1117  bounds_fixed = true;
1118  }
1119  }
1120  for (size_type i = m_lb.size() - m_i_dimension; i < m_lb.size(); ++i) {
1121  // First let's make sure that integer bounds are in the allowed range.
1122  if (m_lb[i] < INT_MIN) {
1123  m_lb[i] = INT_MIN;
1124  bounds_fixed = true;
1125  }
1126  if (m_lb[i] > INT_MAX) {
1127  m_lb[i] = INT_MAX;
1128  bounds_fixed = true;
1129  }
1130  if (m_ub[i] < INT_MIN) {
1131  m_ub[i] = INT_MIN;
1132  bounds_fixed = true;
1133  }
1134  if (m_ub[i] > INT_MAX) {
1135  m_ub[i] = INT_MAX;
1136  bounds_fixed = true;
1137  }
1138  // Then convert them to the nearest integer if necessary.
1139  if (m_lb[i] != double_to_int::convert(m_lb[i])) {
1140  m_lb[i] = double_to_int::convert(m_lb[i]);
1141  bounds_fixed = true;
1142  }
1143  if (m_ub[i] != double_to_int::convert(m_ub[i])) {
1144  m_ub[i] = double_to_int::convert(m_ub[i]);
1145  bounds_fixed = true;
1146  }
1147  }
1148  if (bounds_fixed) {
1149  pagmo_throw(value_error,"problem bounds were invalid and had to be fixed");
1150  }
